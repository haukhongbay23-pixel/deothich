-- 2. HÀM LẤY QUEST (ĐÃ CẬP NHẬT)
function GetQuest()
    local lv = player.Data.Level.Value
    if lv < 15 then return "BanditQuest1", 1
    elseif lv < 30 then return "MonkeyQuest1", 1
    elseif lv < 40 then return "MonkeyQuest1", 2
    elseif lv < 60 then return "PirateIslandQuest1", 1
    elseif lv < 75 then return "PirateIslandQuest1", 2
    elseif lv < 90 then return "DesertQuest", 1
    else return "DesertQuest", 2 end
end

-- 3. LOGIC FARM TỔNG HỢP (FIX REACH, FAST, HAKI, BRING)
task.spawn(function()
    while task.wait() do
        if _G.Farm then
            pcall(function()
                -- Auto Buso
                if not player.Character:FindFirstChild("HasBuso") then
                    game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Buso")
                end
                -- Auto Quest
                if _G.Quest and not player.PlayerGui.Main.Quest.Visible then
                    local q, n = GetQuest()
                    game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", q, n)
                end
                -- Farm Loop
                for _, Monster in pairs(workspace.Enemies:GetChildren()) do
                    if Monster:FindFirstChild("HumanoidRootPart") and Monster.Humanoid.Health > 0 then
                        if _G.Bring then 
                            Monster.HumanoidRootPart.CFrame = player.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, -6)
                            Monster.HumanoidRootPart.CanCollide = false
                        end
                        -- Reach 6.5 unit (Khoảng cách đánh xa an toàn)
                        player.Character.HumanoidRootPart.CFrame = Monster.HumanoidRootPart.CFrame * CFrame.new(0, 6.5, 0)
                        if _G.Fast then
                            player.Character:FindFirstChildOfClass("Tool"):Activate()
                            game:GetService("ReplicatedStorage").Remotes.Validator:FireServer(math.huge)
                        end
                        break 
                    end
                end
            end)
        end
    end
end)

-- 4. LOGIC AUTO CHEST
task.spawn(function()
    while task.wait(0.1) do
        if _G.Chest then
            for _, v in pairs(workspace:GetChildren()) do
                if v.Name:find("Chest") and v:IsA("Part") then
                    player.Character.HumanoidRootPart.CFrame = v.CFrame
                end
            end
        end
    end
end)

-- 5. PHỤ TRỢ (STATS, ESP, FRUIT)
task.spawn(function()
    while task.wait(1) do
        if _G.Stats then game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", 3) end
    end
end)

task.spawn(function()
    while task.wait(0.5) do
        if _G.ESP or _G.AutoFruit then
            for _, v in pairs(game.CoreGui:GetChildren()) do if v.Name == "FruitTrack" then v:Destroy() end end
            for _, v in pairs(workspace:GetChildren()) do
                if v.Name:find("Fruit") or (v:IsA("Tool") and v:FindFirstChild("Handle")) then
                    local target = v:IsA("Tool") and v.Handle or v
                    if _G.ESP then
                        local bill = Instance.new("BillboardGui", game.CoreGui)
                        bill.Name = "FruitTrack"; bill.AlwaysOnTop = true; bill.Size = UDim2.new(0, 100, 0, 50); bill.Adornee = target
                        local txt = Instance.new("TextLabel", bill)
                        txt.Size = UDim2.new(1, 0, 1, 0); txt.Text = "🍎 " .. v.Name; txt.TextColor3 = Color3.fromRGB(255, 0, 0); txt.BackgroundTransparency = 1; txt.Parent = bill
                    end
                    if _G.AutoFruit then player.Character.HumanoidRootPart.CFrame = target.CFrame end
                end
            end
        end
    end
end)

local function SetToggle(btn, var, col)
    _G[var] = not _G[var]
    btn.BackgroundColor3 = _G[var] and col or Color3.fromRGB(30, 30, 30)
end
